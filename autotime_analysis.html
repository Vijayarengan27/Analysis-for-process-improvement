  group_by(PRODUCT_LINE) %>% 
  summarise(parts_demand = mean(Quantity),
            work_demand = mean(Work_orders),
            parts_takt_time = parts_demand/yrly_avail_time,
            work_takt_time =  work_demand/yrly_avail_time)
